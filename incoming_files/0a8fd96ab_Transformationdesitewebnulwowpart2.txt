// ERROR BOUNDARY OBLIGATOIRE (capture TOUS bugs)
class SafeErrorBoundary extends React.Component {
  constructor(props) {
    super(props)
    this.state = { hasError: false, error: null }
  }
  
  static getDerivedStateFromError(error) {
    return { hasError: true, error }
  }
  
  componentDidCatch(error, errorInfo) {
    // Log Sentry/New Relic/Console
    console.error('🚨 COMPONENT CRASH:', error, errorInfo)
    // Analytics event
    sendErrorEvent(error)
  }
  
  render() {
    if (this.state.hasError) {
      return (
        <div className="min-h-screen flex items-center justify-center p-8">
          <div className="max-w-md text-center">
            <h2 className="text-2xl font-bold text-theme-error mb-4">
              Oops... une tuile ! ✨
            </h2>
            <p>Relance la page ou contacte-nous.</p>
            <button 
              onClick={() => window.location.reload()}
              className="mt-4 px-6 py-2 bg-theme-primary text-white rounded-full"
            >
              Recharger la magie
            </button>
          </div>
        </div>
      )
    }
    return this.props.children
  }
}

// BUG DETECTION EN TEMPS RÉEL
const usePerfMonitor = () => {
  const [fps, setFps] = useState(60)
  let lastTime = useRef(0)
  
  useEffect(() => {
    let frameCount = 0
    let lastTimeStamp = 0
    
    const loop = (time) => {
      frameCount++
      
      if (time > lastTimeStamp + 1000) {
        const fps = Math.round((frameCount * 1000) / (time - lastTimeStamp))
        setFps(fps)
        
        if (fps < 50) {
          console.warn('🚨 FPS DROP:', fps)
          // Pause heavy animations
          document.documentElement.style.setProperty(
            '--reduce-motion', 'reduce'
          )
        }
        
        frameCount = 0
        lastTimeStamp = time
      }
      
      rafId.current = requestAnimationFrame(loop)
    }
    
    const rafId = { current: requestAnimationFrame(loop) }
    return () => cancelAnimationFrame(rafId.current)
  }, [])
  
  return fps
}
